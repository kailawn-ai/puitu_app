import { useAuth } from "@/contexts/auth-context";
import { UserService } from "@/lib/services/user-service";
import { useRouter } from "expo-router";
import {
  Briefcase,
  Calendar,
  CheckCircle,
  ChevronLeft,
  ChevronRight,
  Home,
  Mail,
  MapPin,
  Phone,
  User,
} from "lucide-react-native";
import { useColorScheme } from "nativewind";
import React, { useEffect, useState } from "react";
import {
  ActivityIndicator,
  Animated,
  Dimensions,
  KeyboardAvoidingView,
  Platform,
  ScrollView,
  Text,
  TextInput,
  TouchableOpacity,
  View,
} from "react-native";
import { useSafeAreaInsets } from "react-native-safe-area-context";

const { width } = Dimensions.get("window");

interface OnboardingForm {
  name: string;
  phone: string;
  dob: string;
  email: string;
  country: string;
  state: string;
  district: string;
  town: string;
  profile_image: string;
}

export default function OnboardingScreen() {
  const router = useRouter();
  const { user } = useAuth();
  const { colorScheme } = useColorScheme();
  const insets = useSafeAreaInsets();
  const isDark = colorScheme === "dark";

  const [step, setStep] = useState(0);
  const [isLoading, setIsLoading] = useState(false);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [fadeAnim] = useState(new Animated.Value(1));
  const [slideAnim] = useState(new Animated.Value(0));

  const [form, setForm] = useState<OnboardingForm>({
    name: "",
    phone: "",
    dob: "",
    email: "",
    country: "",
    state: "",
    district: "",
    town: "",
    profile_image: "",
  });

  // Prefill data from Firebase/Auth
  useEffect(() => {
    if (user) {
      setForm((prev) => ({
        ...prev,
        name: user.displayName || "",
        email: user.email || "",
        phone: user.phoneNumber || "",
        profile_image: user.photoURL || "",
      }));
    }
  }, [user]);

  // Animate step transitions
  useEffect(() => {
    Animated.parallel([
      Animated.timing(fadeAnim, {
        toValue: 1,
        duration: 300,
        useNativeDriver: true,
      }),
      Animated.spring(slideAnim, {
        toValue: 0,
        tension: 50,
        friction: 7,
        useNativeDriver: true,
      }),
    ]).start();
  }, [step]);

  const handleStepChange = (newStep: number) => {
    Animated.parallel([
      Animated.timing(fadeAnim, {
        toValue: 0,
        duration: 150,
        useNativeDriver: true,
      }),
      Animated.timing(slideAnim, {
        toValue: newStep > step ? 20 : -20,
        duration: 150,
        useNativeDriver: true,
      }),
    ]).start(() => {
      setStep(newStep);
    });
  };

  const updateField = (key: keyof OnboardingForm, value: string) => {
    setForm((prev) => ({ ...prev, [key]: value }));
  };

  const next = () => {
    if (step < totalSteps - 1) {
      handleStepChange(step + 1);
    }
  };

  const back = () => {
    if (step > 0) {
      handleStepChange(step - 1);
    }
  };

  const handleFinish = async () => {
    setIsSubmitting(true);
    try {
      await UserService.updateMe({
        name: form.name,
        phone: form.phone,
        country: form.country,
        state: form.state,
        district: form.district,
        town: form.town,
        dob: form.dob,
      });

      router.replace("/home");
    } catch (e) {
      console.error("Onboarding failed", e);
    } finally {
      setIsSubmitting(false);
    }
  };

  const skip = async () => {
    router.replace("/home");
  };

  const totalSteps = 4;
  const progress = ((step + 1) / totalSteps) * 100;

  const renderStep = () => {
    const animatedStyle = {
      opacity: fadeAnim,
      transform: [{ translateX: slideAnim }],
    };

    switch (step) {
      case 0:
        return (
          <Animated.View style={animatedStyle} className="gap-8">
            <View className="items-center mb-4">
              <View
                className={`w-24 h-24 rounded-2xl bg-primary-100 dark:bg-primary-900/30 items-center justify-center mb-6`}
              >
                <User size={48} color="#7A25FF" />
              </View>
              <Text className="text-3xl font-bold text-secondary-900 dark:text-white text-center">
                Welcome to Puitu! 👋
              </Text>
              <Text className="text-base text-secondary-400 dark:text-secondary-300 text-center mt-2 px-4">
                Lets set up your profile to personalize your experience
              </Text>
            </View>

            <View className="gap-4">
              <InputField
                icon={<User size={20} color={isDark ? "#A3A3A3" : "#737373"} />}
                placeholder="Your full name"
                value={form.name}
                onChangeText={(v) => updateField("name", v)}
                isDark={isDark}
              />
              <InputField
                icon={
                  <Phone size={20} color={isDark ? "#A3A3A3" : "#737373"} />
                }
                placeholder="Phone number"
                value={form.phone}
                onChangeText={(v) => updateField("phone", v)}
                keyboardType="phone-pad"
                isDark={isDark}
              />
              <InputField
                icon={<Mail size={20} color={isDark ? "#A3A3A3" : "#737373"} />}
                placeholder="Email address"
                value={form.email}
                onChangeText={(v) => updateField("email", v)}
                keyboardType="email-address"
                isDark={isDark}
                editable={false}
              />
            </View>
          </Animated.View>
        );

      case 1:
        return (
          <Animated.View style={animatedStyle} className="gap-8">
            <View className="items-center mb-4">
              <View
                className={`w-24 h-24 rounded-2xl bg-primary-100 dark:bg-primary-900/30 items-center justify-center mb-6`}
              >
                <MapPin size={48} color="#7A25FF" />
              </View>
              <Text className="text-3xl font-bold text-secondary-900 dark:text-white text-center">
                Where are you from? 📍
              </Text>
              <Text className="text-base text-secondary-400 dark:text-secondary-300 text-center mt-2 px-4">
                Help us connect you with local farmers
              </Text>
            </View>

            <View className="gap-4">
              <InputField
                icon={
                  <MapPin size={20} color={isDark ? "#A3A3A3" : "#737373"} />
                }
                placeholder="Country"
                value={form.country}
                onChangeText={(v) => updateField("country", v)}
                isDark={isDark}
              />
              <InputField
                icon={
                  <MapPin size={20} color={isDark ? "#A3A3A3" : "#737373"} />
                }
                placeholder="State"
                value={form.state}
                onChangeText={(v) => updateField("state", v)}
                isDark={isDark}
              />
              <InputField
                icon={
                  <MapPin size={20} color={isDark ? "#A3A3A3" : "#737373"} />
                }
                placeholder="District"
                value={form.district}
                onChangeText={(v) => updateField("district", v)}
                isDark={isDark}
              />
              <InputField
                icon={<Home size={20} color={isDark ? "#A3A3A3" : "#737373"} />}
                placeholder="Town/City"
                value={form.town}
                onChangeText={(v) => updateField("town", v)}
                isDark={isDark}
              />
            </View>
          </Animated.View>
        );

      case 2:
        return (
          <Animated.View style={animatedStyle} className="gap-8">
            <View className="items-center mb-4">
              <View
                className={`w-24 h-24 rounded-2xl bg-primary-100 dark:bg-primary-900/30 items-center justify-center mb-6`}
              >
                <Briefcase size={48} color="#7A25FF" />
              </View>
              <Text className="text-3xl font-bold text-secondary-900 dark:text-white text-center">
                Tell us about yourself 🌾
              </Text>
              <Text className="text-base text-secondary-400 dark:text-secondary-300 text-center mt-2 px-4">
                This helps us recommend relevant content
              </Text>
            </View>

            <View className="gap-4">
              <InputField
                icon={
                  <Calendar size={20} color={isDark ? "#A3A3A3" : "#737373"} />
                }
                placeholder="Date of birth (DD/MM/YYYY)"
                value={form.dob}
                onChangeText={(v) => updateField("dob", v)}
                isDark={isDark}
              />
            </View>
          </Animated.View>
        );

      case 3:
        return (
          <Animated.View style={animatedStyle} className="gap-8">
            <View className="items-center mb-4">
              <View
                className={`w-24 h-24 rounded-2xl bg-green-100 dark:bg-green-900/30 items-center justify-center mb-6`}
              >
                <CheckCircle size={48} color="#10B981" />
              </View>
              <Text className="text-3xl font-bold text-secondary-900 dark:text-white text-center">
                Youre all set! 🎉
              </Text>
              <Text className="text-base text-secondary-400 dark:text-secondary-300 text-center mt-2 px-4">
                Your profile is ready to go. Lets start exploring!
              </Text>
            </View>

            <View className="bg-primary-50 dark:bg-primary-900/20 p-6 rounded-2xl">
              <Text className="text-lg font-semibold text-secondary-900 dark:text-white mb-3">
                Profile Summary
              </Text>
              <SummaryRow label="Name" value={form.name} isDark={isDark} />
              <SummaryRow
                label="Location"
                value={`${form.town}, ${form.district}`}
                isDark={isDark}
              />
            </View>
          </Animated.View>
        );

      default:
        return null;
    }
  };

  if (isLoading) {
    return (
      <View
        className="flex-1 items-center justify-center bg-white dark:bg-background-ddark"
        style={{
          paddingTop: insets.top,
          paddingBottom: insets.bottom,
          paddingLeft: insets.left,
          paddingRight: insets.right,
        }}
      >
        <ActivityIndicator size="large" color="#7A25FF" />
      </View>
    );
  }

  return (
    <View
      className="flex-1 bg-white dark:bg-background-ddark"
      style={{
        paddingTop: insets.top,
        paddingBottom: insets.bottom,
        paddingLeft: insets.left,
        paddingRight: insets.right,
      }}
    >
      <KeyboardAvoidingView
        style={{ flex: 1 }}
        behavior={Platform.OS === "ios" ? "padding" : "height"}
        keyboardVerticalOffset={Platform.OS === "ios" ? 0 : 20}
      >
        {/* Header with Progress */}
        <View className="px-6 pt-2">
          <View className="flex-row justify-between items-center mb-4">
            <TouchableOpacity
              onPress={back}
              disabled={step === 0}
              hitSlop={{ top: 10, bottom: 10, left: 10, right: 10 }}
            >
              <ChevronLeft
                size={24}
                color={
                  step === 0
                    ? isDark
                      ? "#404040"
                      : "#D4D4D4"
                    : isDark
                      ? "#FFFFFF"
                      : "#09090b"
                }
              />
            </TouchableOpacity>
            <Text className="text-secondary-500 dark:text-secondary-400 font-medium">
              Step {step + 1} of {totalSteps}
            </Text>
            <View style={{ width: 24 }} /> {/* Placeholder for balance */}
          </View>

          {/* Progress Bar */}
          <View className="h-2 bg-secondary-100 dark:bg-secondary-800 rounded-full overflow-hidden">
            <View
              className="h-full bg-primary-500 rounded-full"
              style={{ width: `${progress}%` }}
            />
          </View>
        </View>

        <ScrollView
          className="flex-1"
          contentContainerStyle={{
            flexGrow: 1,
            paddingHorizontal: 24,
            paddingTop: 16,
            paddingBottom: 24,
          }}
          showsVerticalScrollIndicator={false}
          keyboardShouldPersistTaps="handled"
          bounces={false}
        >
          {renderStep()}
        </ScrollView>

        {/* Footer */}
        <View className="px-6 pb-2 pt-4 border-t border-border dark:border-border-dark bg-white dark:bg-background-ddark">
          <View className="flex-row gap-3">
            {step > 0 && (
              <TouchableOpacity
                onPress={back}
                className="py-4 rounded-2xl border border-border dark:border-border-dark flex-row items-center justify-center px-5"
                activeOpacity={0.7}
              >
                <ChevronLeft size={20} color={isDark ? "#A3A3A3" : "#737373"} />
              </TouchableOpacity>
            )}
            {step < totalSteps - 1 && (
              <TouchableOpacity
                onPress={skip}
                className="flex-1 py-4 rounded-2xl border border-border dark:border-border-dark"
                activeOpacity={0.7}
              >
                <Text className="text-center text-secondary-600 dark:text-secondary-300 font-semibold">
                  Skip
                </Text>
              </TouchableOpacity>
            )}

            {step < totalSteps - 1 ? (
              <TouchableOpacity
                onPress={next}
                className="flex-1 py-4 rounded-2xl bg-primary-500 shadow-button flex-row items-center justify-center"
                activeOpacity={0.7}
              >
                <Text className="text-white font-semibold text-base mr-2">
                  Next
                </Text>
                <ChevronRight size={20} color="white" />
              </TouchableOpacity>
            ) : (
              <TouchableOpacity
                onPress={handleFinish}
                disabled={isSubmitting}
                className={`flex-1 py-4 rounded-2xl bg-primary-500 shadow-button flex-row items-center justify-center ${isSubmitting ? "opacity-70" : ""}`}
                activeOpacity={0.7}
              >
                {isSubmitting ? (
                  <ActivityIndicator color="white" size="small" />
                ) : (
                  <>
                    <Text className="text-white font-semibold text-base mr-2">
                      Finish
                    </Text>
                    <CheckCircle size={20} color="white" />
                  </>
                )}
              </TouchableOpacity>
            )}
          </View>
        </View>
      </KeyboardAvoidingView>
    </View>
  );
}

// Reusable Input Field Component
interface InputFieldProps {
  icon: React.ReactNode;
  placeholder: string;
  value: string;
  onChangeText: (text: string) => void;
  keyboardType?: "default" | "email-address" | "numeric" | "phone-pad";
  isDark: boolean;
  editable?: boolean;
}

const InputField: React.FC<InputFieldProps> = ({
  icon,
  placeholder,
  value,
  onChangeText,
  keyboardType = "default",
  isDark,
  editable = true,
}) => (
  <View
    className={`flex-row items-center bg-white dark:bg-secondary-800 rounded-2xl border ${isDark ? "border-border-dark" : "border-border"} px-4 ${!editable ? "opacity-60" : ""}`}
  >
    <View className="mr-3">{icon}</View>
    <TextInput
      placeholder={placeholder}
      className="flex-1 py-4 text-base text-secondary-900 dark:text-white"
      placeholderTextColor={isDark ? "#737373" : "#A3A3A3"}
      value={value}
      onChangeText={onChangeText}
      keyboardType={keyboardType}
      editable={editable}
    />
  </View>
);

// Summary Row Component
const SummaryRow: React.FC<{
  label: string;
  value: string;
  isDark: boolean;
}> = ({ label, value, isDark }) => (
  <View className="flex-row justify-between py-2 border-b border-border dark:border-border-dark">
    <Text className="text-secondary-500 dark:text-secondary-400">{label}</Text>
    <Text className="text-secondary-900 dark:text-white font-medium">
      {value || "Not provided"}
    </Text>
  </View>
);
